        /**  Stream Abstraction
        * @class
        * @classdesc This class scan an input stream and convert it to an token input */
        var stream = (function()
        {
            function stream (inpt)
            {
                this.inpt = inpt;
            }
            
            stream.prototype.getAll = function() {
                return inpt;
            }
            
            return stream;
        })();
